PACKAGE github.com/faustbrian/golib/pkg/http-middleware
package middleware // import "github.com/faustbrian/golib/pkg/http-middleware"

Package middleware composes explicit net/http middleware without a registry.

Middleware are listed in request execution order. Responses unwind through that
list in reverse order.

CONSTANTS

const MaxChainDepth = 256
    MaxChainDepth bounds composition work and nested handler depth.

	ErrNilPredicate = errors.New("middleware: nil predicate")
	// ErrInvalidName identifies an empty or unsafe descriptor name.

    Middleware is the standard net/http handler decorator contract.

func When(predicate Predicate, middleware Middleware) (Middleware, error)
    When constructs middleware that evaluates predicate once per request. The

    A panic propagates to the caller; cancellation is observable through

type Policy struct{ Timeout time.Duration }
    Policy configures a handler context timeout of at most 24 hours. Code that
    ignores its context is not interrupted.

type TimeoutPolicy struct {
	Timeout          time.Duration
	MaxResponseBytes int
	MaxConcurrent    int
	Status           int
}
    TimeoutPolicy configures a bounded non-streaming handler timeout of at
    most 24 hours. MaxConcurrent defaults to 1024 and is capped at 65,536.
    A handler that ignores cancellation may continue executing and retain one
    concurrency slot, but cannot write through the closed buffer or create
    unbounded middleware-owned goroutine growth.

type Event struct {

	Route       string

	Bytes       int64
	Duration    time.Duration
	Proto       string
	Outcome     Outcome
	ClientClass string
}
    Event excludes raw paths, queries, headers, payloads, identities, and

type Outcome string
    Outcome is a bounded completion classification.

const (
	// Success classifies responses below 400.
	Success Outcome = "success"
	// ClientError classifies responses from 400 through 499.
	ClientError Outcome = "client_error"
	// ServerError classifies responses at or above 500.
	ServerError Outcome = "server_error"
	// Canceled classifies requests whose context was canceled.
	Canceled Outcome = "canceled"

	Panicked Outcome = "panicked"
)

	Route           func(*http.Request) string
	ClientClass     func(*http.Request) string
	Now             func() time.Time
	RepanicObserver bool
}
    Policy injects observation without owning a logger or telemetry SDK.
    Route and ClientClass run once at completion; their panics are contained.
